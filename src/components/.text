// components/FontComponents.tsx
import { ReactNode } from 'react';

interface TextProps {
  children: ReactNode;
  className?: string;
  as?: keyof JSX.IntrinsicElements;
}

// Base Poppins component
export function PoppinsText({
  children,
  className = "",
  as: Component = "span"
}: TextProps & { as?: keyof JSX.IntrinsicElements }) {
  return (
    <Component
      style={{ fontFamily: 'var(--font-poppins)' }}
      className={className}
    >
      {children}
    </Component>
  );
}

// Specific weight components
export function PoppinsThin({ children, className = "", as = "span" }: TextProps) {
  return (
    <PoppinsText as={as} className={`font-thin ${className}`}>
      {children}
    </PoppinsText>
  );
}

export function PoppinsLight({ children, className = "", as = "span" }: TextProps) {

}
